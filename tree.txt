
#include <iostream>
#include<stdlib.h>
using namespace std;
struct node
{
int data;
node *left;
node *right;
};
node *create_node(int n)
{
node *root=new node();
root->data=n;
root->left=NULL;
root->right=NULL;
return root;
}
struct node* insert(node *root,int n)
{
if(root==NULL)
{
root=create_node(n);
}
else if(n<=root->data)
{
root->left=insert(root->left,n);
}
else
{
root->right=insert(root->right,n);
}
return root;
}
void preorder(node *root)
{
if(root)
{
cout<<root->data<<" ";
preorder(root->left);
preorder(root->right);

}
}
void inorder(node *root)
{
if(root)
{
inorder(root->left);
cout<<root->data<<" ";
inorder(root->right);
}
}
void postorder(node *root)
{
if(root)
{
postorder(root->left);
postorder(root->right);
cout<<root->data<<" ";
}
}
int main()
{
int n;
int choice;
node *root=NULL;
cout<<"press 1 for insertion "<<endl;
cout<<"press 2 for pre-order "<<endl;
cout<<"press 3 for in-order"<<endl;
cout<<"press 4 for postorder"<<endl;
cout<<"press 5 for stop execution "<<endl;
while(true)
{
cout<<"\n press your choice"<<endl;
cin>>choice;
switch(choice)
{
case 1:
cout<<"insert a number to be insert"<<endl;
cin>>n;
root=insert(root,n);

break;
case 2:
preorder(root);
break;
case 3:
inorder(root);
break;
case 4:
postorder(root);
break;
case 5:
exit(0);
break;
default:
cout<<"wrong choice"<<endl;
}
}
}
